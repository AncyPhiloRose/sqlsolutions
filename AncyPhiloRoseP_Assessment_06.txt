1) SELECT E.EMP_ID,E.EMP_NAME,D.DESC_NAME,P.PLACE_DESC,E.EMP_DESC_CNT,P.EMP_PLACE_CNT
   CASE WHEN P.PLACE_DESC IS NULL OR WHEN D.DESC_NAME IS NULL THEN 'OTHERS'
   ELSE D.DESC_NAME,P.PLACE_DESC END AS E.EMP_DESC_CNT,P.EMP_PLACE_CNT
   FROM EMPLOYEE E LEFT JOIN DEIGNATION D ON E.DESC_ID=D.DESC_ID LEFT JOIN
   PLACE P ON P.PLACE_ID=E.PLACE_ID LEFT JOIN PROJECT PR ON PR.PROJECT_ID=E.PROJECT_ID;

2)SELECT E.EMP_ID,E.EMP_NAME,D.DESC_NAME,P.PLACE_DESC,PR.PROJ_NAME 
 FROM EMPLOYEE E LEFT JOIN DESTINATION D ON E.DESC_ID=D.DESC_ID LEFT JOIN PLACE P ON P.PLACE_ID=E.PLACE_ID
 LEFT JOIN PROJECT PR ON PR.PROJECT_ID=E.PROJECT_ID WHERE PROJECT_NAME='CLOUD' OR PROJECT_NAME='DATA';
